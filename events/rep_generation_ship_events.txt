namespace = repopulate_mod

# Generation Ship Events

repopulation_enable_generation_ships = {
	type = country_event
	placement = root

	title = repopulate_mod.1.t
	desc = repopulate_mod.1.d
	flavor = repopulate_mod.1.f

	duration = 1

	event_image = {
		video = repopulate_generation_ship_video
	}

	icon = "gfx/interface/icons/generation_ship.dds"

	trigger = {
		NOT = {
			has_modifier = repopulation_generation_ships_landed
		}
	}

	immediate = {}

	option = { 
		name = repopulate_mod.1.a

		add_modifier = {
			name = repopulation_generation_ships_landed
			days = never_fire_again_modifier_time
		}
		
		add_modifier = {
			name = repopulation_generation_ships_enabled
			days = 10 * 365
		}
	}
}
